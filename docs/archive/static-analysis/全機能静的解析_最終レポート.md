   ```rust
       message: format!("タスク {:?} は無効になっています", task_type),
       code: "TASK_DISABLED".to_string(),
   }
   
   // 修正必要
   AppError::ApiError {
       message: format!("タスク {:?} は無効になっています", task_type),
       code: "TASK_DISABLED".to_string(),
       source: None,
   }
   ```
